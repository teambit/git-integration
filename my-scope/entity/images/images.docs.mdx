This component describes the Image object. It serves as a contract between components providing and consuming Images.

### Creating a new image object

```ts
import { Images } from '@teambit/community.entity.images';

const alt = 'ebay';
const src = 'https://static.bit.dev/homepage-enterprise-logos/ebay.svg';
const loading = 'lazy';

const image = new Images(alt, src, loading);
```

### Consuming images via an API

```ts
import type { ImagesFromApi } from '@teambit/community.entity.images';
import { Images } from '@teambit/community.entity.images';
import { fetchImage } from './fetch-image';

async () => {
  const fetchedImage: ImagesFromApi = await fetchImage();
  const image = Images.fromApiObject(fetchedImage);
};
```
